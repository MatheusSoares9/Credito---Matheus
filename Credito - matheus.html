#include <stdio.h>
#include <string.h>
#include <locale.h>

int main() {
	
	setlocale(LC_ALL,"Portuguese");

	
    float valor_compra, total_compras = 0.0;
    char ultima_compra;
    int dia_pagamento;
    
    do {
        printf("Informe o valor da compra: R$ ");
        scanf("%f", &valor_compra);
        
    
        total_compras += valor_compra;
        
        
        printf("Essa é a última compra? (s/n):  ");
        scanf(" %c", &ultima_compra);
        
    } while (ultima_compra != 's');


    printf("Informe o dia do pagamento da fatura: ");
    scanf("%d", &dia_pagamento);

    
    float juros = 0.0;

    
    if (dia_pagamento > 25) {
        int dias_atraso = dia_pagamento - 25;F
        
        juros = total_compras * 0.03 + total_compras * 0.01 * dias_atraso;
    }

    
    	float total_a_pagar = total_compras + juros;

    printf("\nResumo da fatura:\n");
    printf("Total das compras: R$ %.2f\n", total_compras);
    printf("Juros: R$ %.2f\n", juros);
    printf("Total a pagar: R$ %.2f\n", total_a_pagar);

    return 0;
}
